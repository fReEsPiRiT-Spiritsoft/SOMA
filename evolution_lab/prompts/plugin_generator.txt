"""Plugin: System-Info"""
__version__ = "0.1.0"
__author__ = "soma-ai"
__description__ = "Zeigt CPU/RAM-Auslastung"
__dependencies__ = ["psutil"]

import asyncio
import logging
import psutil

logger = logging.getLogger("soma.plugin.sysinfo")

async def on_load() -> None:
    logger.info("sysinfo_loaded")

async def execute(query: str = "all") -> str:
    try:
        cpu = psutil.cpu_percent(interval=0.5)
        ram = psutil.virtual_memory()
        return f"CPU: {cpu}%  RAM: {ram.percent}% ({ram.used // 1024**3}GB / {ram.total // 1024**3}GB)"
    except Exception as e:
        return f"Fehler: {e}"

async def on_unload() -> None:
    pass

"""Plugin: Bluetooth Status"""
__version__ = "0.1.0"
__author__ = "soma-ai"
__description__ = "Prüft Bluetooth-Status via bluetoothctl"

import asyncio
import logging

logger = logging.getLogger("soma.plugin.bluetooth")

async def on_load() -> None:
    logger.info("bluetooth_plugin_loaded")

async def execute(command: str = "status") -> str:
    try:
        proc = await asyncio.create_subprocess_exec(
            "bluetoothctl", "show",
            stdout=asyncio.subprocess.PIPE,
            stderr=asyncio.subprocess.PIPE,
        )
        stdout, stderr = await proc.communicate()
        if proc.returncode != 0:
            return f"Fehler: {stderr.decode().strip()}"
        output = stdout.decode().strip()
        powered = "powered: yes" in output.lower()
        return "Bluetooth ist eingeschaltet." if powered else "Bluetooth ist ausgeschaltet."
    except FileNotFoundError:
        return "bluetoothctl nicht gefunden."
    except Exception as e:
        return f"Fehler: {e}"

async def on_unload() -> None:
    pass
